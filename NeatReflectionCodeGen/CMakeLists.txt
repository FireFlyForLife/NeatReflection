# Third party

	docopt
	GIT_REPOSITORY https://github.com/docopt/docopt.cpp.git
	GIT_TAG v0.6.3
)

	ifc-reader
	GIT_REPOSITORY https://github.com/AndreyG/ifc-reader.git

)
